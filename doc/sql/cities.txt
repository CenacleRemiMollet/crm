-- Around Bry sur Marne (48.833333, 2.516667)

SELECT id, city_name, (
    6371 * acos (
      cos ( radians(48.833333) )
      * cos( radians( latitude ) )
      * cos( radians( longitude ) - radians(2.516667) )
      + sin ( radians(48.833333) )
      * sin( radians( latitude ) )
    )
  ) AS distance
 FROM city
 HAVING distance < 5
 ORDER BY distance;
